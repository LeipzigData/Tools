PREFIX ld: <http://leipzig-data.de/Data/Model/> 
PREFIX geosparql: <http://www.opengis.net/ont/geosparql#>
PREFIX ki: <http://kultur-initiative.net/Data/Model#>
select ?a ?u ?l ?g ?g1
from <http://leipzig-data.de/Data/Adressen/>
from <http://leipzig-data.de/Data/GeoDaten/>
from <http://kultur-initiative.net/Data/Akteure-1/>
where {
?a a ld:Akteur; ld:hasAddress ?u .
?u rdfs:label ?l .
optional { ?u geosparql:asWKT ?g .}
optional { ?a ki:hasGeo ?g1 .}
}

PREFIX ld: <http://leipzig-data.de/Data/Model/> 
construct { ?u ?v ?w .}
from <http://leipzig-data.de/Data/Adressen/>
from <http://leipzig-data.de/Data/GeoDaten/>
from <http://kultur-initiative.net/Data/Akteure-1/>
where {
?a a ld:Akteur; ld:hasAddress ?u .
?u ?v ?w .
}
