Ein paar Beispiel-Anfragen:

Alle APIRefs finden

SELECT distinct ?p ?pn WHERE {
  ?pn ld:hasAPIRef ?p . 
} 

--------------------------------------------

Alle Events nach einem bestimmten Datum finden

SELECT distinct ?a ?as WHERE {
  ?a a ld:Event  . 
  ?a cal:dtstart ?as . 
FILTER ( xsd:dateTime(?as) > xsd:dateTime("2012-01-01T00:00:00Z") )
} 

--------------------------------------------

Alle Orte finden, die in Events als cal:location vorkommen

SELECT distinct ?e ?o ?ol WHERE {
  ?e a ld:Event .
  ?e cal:location ?o .
  optional { ?o rdfs:label ?ol .}
} 

Alle Orte finden, die keinen Label haben

SELECT distinct ?o ?ol WHERE {
  ?o a ld:Ort ;
  optional { ?o rdfs:label ?ol .}
  filter (!bound(?ol))
} 

SELECT distinct ?e WHERE {
  ?e a ld:Event .
  ?e cal:location <http://leipzig-data.de/Data/Ort/KreativesLeipzig> .
} 

--------------------------------------------

Alle Datensätze mit Prädikat ?p finden, das "zusatz" case insensitiv matcht.

select ?a ?p ?b where {
?a ?p ?b .
  filter regex(?p,"zusatz","i") .
}

select ?a ?p ?b where {
?a ?p ?b .
  filter regex(?p,"Adresse","i") .
}

Wie ist Hs 15 an einem Tag belegt?

SELECT ?lv ?t WHERE {
  ?lv od:locatedAt odp:Hs_15.
  ?lv od:dayOfWeek "donnerstags".
  ?lv od:beginsAt ?t.
} ;

Studenten in einem Studiengang

SELECT ?s ?e WHERE {
  ?s od:estimatedPeople ?e.
  ?s od:hasSemesterType "W".
} ORDER BY (?s);

estimatedPeople zu einer Unit

SELECT ?s ?e WHERE {
  <http://od.fmi.uni-leipzig.de/studium/DB.DBS-2> od:recommendedFor ?s .
  ?s od:estimatedPeople ?e.
  ?s od:hasSemesterType "S".
};

SELECT ?s WHERE {
  ?s a od:Unit .
  FILTER regex(str(?s), "BIS")
};

SELECT ?s ?m ?l WHERE {
  ?s a od:Unit . optional { 
  ?s od:relatedModule ?m. 
  ?m rdfs:label ?l. }
};

  ?u od:containsKurs ?k .

LV eines Studiengangs

SELECT distinct ?lvn ?d ?t WHERE {
 { ?b od:recommendedFor ods:Inf.Bachelor.5 .
  ?b od:containsLV ?lv.
  ?lv rdfs:label ?lvn. }
 optional { ?lv od:dayOfWeek ?d.
 ?lv od:beginsAt ?t.}
}  order by desc(?d);

Meldung zu einer bestimmten LV an Frau Pohl

SELECT ?d  ?b ?e ?r ?p WHERE {
  <http://od.fmi.uni-leipzig.de/w11/LA.GW-Math> od:containsLV ?lv .
  ?lv rdf:type od:Uebung .
  ?lv od:servedBy ?pn .
  ?pn rdfs:label ?p .
  ?lv od:dayOfWeek ?d .
  ?lv od:beginsAt ?b . 
  ?lv od:endsAt ?e .
  ?lv od:locatedAt ?rn .
  ?rn rdfs:label ?r .
} ;

Welche Personen haben welche LV? 

SELECT ?id ?lv WHERE {
  ?id a od:Person .
  ?lv od:servedBy ?id .
} ;

Liste der od:relatedUnit

SELECT ?b ?u WHERE {
  ?b od:relatedUnit ?u .
} ;

Welche Personen halten in welchem Block welche Vorlesung? 

SELECT ?b ?pn ?lvn WHERE {
  ?p a od:Person .
  ?lv od:servedBy ?p .
  ?p rdfs:label ?pn .
  ?lv rdfs:label ?lvn .
  ?lv a od:Vorlesung.
  ?b od:containsLV ?lv .
} ;

Welche LV-Kräfte sind kein od:Person?

SELECT distinct ?id ?t WHERE {
  optional {?id a ?t .} 
  {?lv od:servedBy ?id .} 
  filter (!bound(?t))
} ;

Welche Units sind ohne Modul ?

SELECT distinct ?id WHERE {
  optional {?id od:recommendedFor ?t .} 
  {?id a od:Unit .} 
  filter (!bound(?t))
} ;

Welche Module sind ohne Unit ?

SELECT distinct ?t WHERE {
  optional {?id od:relatedModule ?t .} 
  {?t a od:Module .} 
  filter (!bound(?id))
} ;

Welche LV haben keinen Kurs  ?

SELECT distinct ?lv WHERE {

  {?lv a od:LV .} 
  filter (!bound(?c))
} ;

Welche Units eines Semestertyps zu einem Level gibt es? 

SELECT ?u ?e WHERE { {
  ?u a od:Unit .
  ?u od:recommendedFor ?s .
  ?s od:hasSemesterType "W" .
  ?u od:hasLevel "1" . } optional {
  ?u od:estimatedPeople ?e . }
} ;

Welche Lehrveranstaltungen haben ein od:additionalInformation Eintrag? 

SELECT ?lv ?tag WHERE {
  ?lv a od:LV .
  ?lv od:additionalInformation ?tag .
} ;

Komplette Belegungsliste von Hs 15

SELECT ?d ?b ?e ?t ?p WHERE {
  { ?lv od:locatedAt odr:Hs_15. } optional {
  ?lv od:dayOfWeek ?d.
  ?lv rdfs:label ?t. 
  ?lv od:servedBy ?pn. 
  ?pn rdfs:label ?p. 
  ?lv od:beginsAt ?b. 
  ?lv od:endsAt ?e. }
} 
ORDER BY (?d,?b);

Komplette Belegungsliste überhaupt

SELECT ?r ?d ?b ?e ?t ?p WHERE {
  { ?lv od:locatedAt ?rn. 
    ?rn rdfs:label ?r.
  } optional {
  ?lv od:dayOfWeek ?d.
  ?lv rdfs:label ?t. 
  ?lv od:servedBy ?pn. 
  ?pn rdfs:label ?p. 
  ?lv od:beginsAt ?b. 
  ?lv od:endsAt ?e. }
} 
ORDER BY (?r,?d,?b);

Welche Blöcke mit Übungen gibt es?

  ?u a od:Uebung . 
} ;

-- alle Module, die zum Studiengang Bachelor Informatik gehören

SELECT distinct ?b ?bn WHERE {
  ?b a od:Module .
  ?b od:toStudiengangSemester ?bs .
  ?bs od:toStudiengang <http://od.fmi.uni-leipzig.de/studium/Inf.Bachelor> .
  ?b rdfs:label ?bn .
} order by (?b);

-- alle Studiengänge, die Units verwenden, die auch im Studiengang Bachelor
   Informatik verwendet werden. 

SELECT distinct ?s ?u WHERE {
  ?u a od:Unit .
  ?u od:recommendedFor ?bs .
  ?u od:recommendedFor ?s .
  ?bs od:toStudiengang <http://od.fmi.uni-leipzig.de/studium/Inf.Bachelor> .
} order by (?s);

-- alle Blöcke und ihre zugehörige Unit, so weit vorhanden

SELECT distinct ?b ?u  WHERE {
  ?b a od:Block . optional {
  ?b od:relatedUnit ?u . }
} order by (?b);

-- alle Units eines gegebenen Levels in einem Semestertyp

SELECT distinct ?u  WHERE {
  ?u a od:Unit . 
  ?u od:hasLevel "1" .
  ?u od:recommendedFor ?s .
  ?s od:hasSemesterType "S" .  
} ;
